labelSize(string s) =>
    s == SZ_TINY ? size.tiny : s == SZ_NORMAL ? size.normal : size.small

lineStyle(string s) =>
    s == DASHED ? line.style_dashed : s == DOTTED ? line.style_dotted : line.style_solid

// Color faded by trend-strength (break magnitude relative to ATR).
// Repaint-safe: derived only from already-known data on the current bar.
strengthCol(color base, float magnitude) =>
    if not gradientTrendInp
        base
    else
        float r = math.min(magnitude / math.max(atrMeasure, 1e-10), 3.0) / 3.0
        color.from_gradient(r, 0.0, 1.0, color.new(base, 55), color.new(base, 0))

// Leg detection: 0 = bearish leg, 1 = bullish leg
leg(int size) =>
    var int legState = 0
    bool newHigh = high[size] > ta.highest(size)
    bool newLow  = low[size]  < ta.lowest(size)
    if newHigh
        legState := BEARISH_LEG
    else if newLow
        legState := BULLISH_LEG
    legState

startOfNewLeg(int legV)     => ta.change(legV) != 0
startOfBearishLeg(int legV) => ta.change(legV) == -1
startOfBullishLeg(int legV) => ta.change(legV) == +1

drawLabel(int labelTime, float labelPrice, string tag, color textCol, string style) =>
    var label l = na
    if modeInp == PRESENT
        l.delete()
    l := label.new(chart.point.new(labelTime, na, labelPrice), tag, xloc=xloc.bar_time,
         color=color(na), textcolor=textCol, style=style, size=size.small)

drawEqualHighLow(pivot p, float level, int size, bool isHigh) =>
    equalDisplay disp = isHigh ? equalHighDisplay : equalLowDisplay
    string tag        = isHigh ? 'EQH' : 'EQL'
    color  col        = isHigh ? swBearCol : swBullCol
    string lblStyle   = isHigh ? label.style_label_down : label.style_label_up

    if modeInp == PRESENT
        line.delete(disp.l_ine)
        label.delete(disp.l_abel)

    disp.l_ine := line.new(
         chart.point.new(p.barTime, na, p.currentLevel),
         chart.point.new(time[size], na, level),
         xloc=xloc.bar_time, color=col, style=line.style_dotted)
    int lblPos = math.round(0.5 * (p.barIndex + bar_index - size))
    disp.l_abel := label.new(chart.point.new(na, lblPos, level), tag, xloc=xloc.bar_index,
         color=color(na), textcolor=col, style=lblStyle, size=labelSize(eqLblSzInp))

// ════════════════════════════════════════════════════════════════════════════
// STRUCTURE: PIVOT DETECTION & STORAGE
// ════════════════════════════════════════════════════════════════════════════
getCurrentStructure(int size, bool equalHL = false, bool internal = false) =>
    int  legV     = leg(size)
    bool newPivot = startOfNewLeg(legV)
    bool isLow    = startOfBullishLeg(legV)
    bool isHigh   = startOfBearishLeg(legV)

    if newPivot
        if isLow
            pivot p = equalHL ? equalLow : internal ? internalLow : swingLow
            if equalHL and math.abs(p.currentLevel - low[size]) < eqThreshInp * atrMeasure
                drawEqualHighLow(p, low[size], size, false)
                currentAlerts.eqLows := true
            p.lastLevel    := p.currentLevel
            p.currentLevel := low[size]
            p.crossed      := false
            p.barTime      := time[size]
            p.barIndex     := bar_index[size]
            if not equalHL and not internal
                trailing.bottom         := p.currentLevel
                trailing.barTime        := p.barTime
                trailing.barIndex       := p.barIndex
                trailing.lastBottomTime := p.barTime
            if showSwingsInp and not internal and not equalHL
                drawLabel(time[size], p.currentLevel,
                     p.currentLevel < p.lastLevel ? 'LL' : 'HL',
                     swBullCol, label.style_label_up)
        else
            pivot p = equalHL ? equalHigh : internal ? internalHigh : swingHigh
            if equalHL and math.abs(p.currentLevel - high[size]) < eqThreshInp * atrMeasure
                drawEqualHighLow(p, high[size], size, true)
                currentAlerts.eqHighs := true
            p.lastLevel    := p.currentLevel
            p.currentLevel := high[size]
            p.crossed      := false
            p.barTime      := time[size]
            p.barIndex     := bar_index[size]
            if not equalHL and not internal
                trailing.top         := p.currentLevel
                trailing.barTime     := p.barTime
                trailing.barIndex    := p.barIndex
                trailing.lastTopTime := p.barTime
            if showSwingsInp and not internal and not equalHL
                drawLabel(time[size], p.currentLevel,
                     p.currentLevel > p.lastLevel ? 'HH' : 'LH',
                     swBearCol, label.style_label_down)

// ════════════════════════════════════════════════════════════════════════════
// STRUCTURE DRAWING
// ════════════════════════════════════════════════════════════════════════════
drawStructure(pivot p, string tag, color structCol, string lStyle, string lblStyle, string lblSize) =>
    var line  l = line.new(na,na,na,na, xloc=xloc.bar_time)
    var label b = label.new(na,na)
    if modeInp == PRESENT
        l.delete()
        b.delete()
    l := line.new(
         chart.point.new(p.barTime, na, p.currentLevel),
         chart.point.new(time, na, p.currentLevel),
         xloc=xloc.bar_time, color=structCol, style=lStyle)
    b := label.new(
         chart.point.new(na, math.round(0.5 * (p.barIndex + bar_index)), p.currentLevel),
         tag, xloc=xloc.bar_index, color=color(na), textcolor=structCol,
         style=lblStyle, size=lblSize)

// ORDER BLOCKS
// ════════════════════════════════════════════════════════════════════════════
deleteOrderBlocks(bool internal = false) =>
    array<orderBlock> obs = internal ? internalOrderBlocks : swingOrderBlocks
    for [idx, ob] in obs
        bool crossed = false
        if bearMitSrc > ob.barHigh and ob.bias == BEARISH
            crossed := true
            if internal
                currentAlerts.intBearOBMit := true
            else
                currentAlerts.swBearOBMit := true
        else if bullMitSrc < ob.barLow and ob.bias == BULLISH
            crossed := true
            if internal
                currentAlerts.intBullOBMit := true
            else
                currentAlerts.swBullOBMit := true
        if crossed
            obs.remove(idx)

storeOrderBlock(pivot p, bool internal, int bias) =>
    if (not internal and showSwOBInp) or (internal and showIntOBInp)
        array<float> arr = na
        int idx = na
        if bias == BEARISH
            arr := parsedHighs.slice(p.barIndex, bar_index)
            idx := p.barIndex + arr.indexof(arr.max())
        else
            arr := parsedLows.slice(p.barIndex, bar_index)
            idx := p.barIndex + arr.indexof(arr.min())
        orderBlock ob = orderBlock.new(parsedHighs.get(idx), parsedLows.get(idx), times.get(idx), bias)
        array<orderBlock> obs = internal ? internalOrderBlocks : swingOrderBlocks
        if obs.size() >= 100
            obs.pop()
        obs.unshift(ob)

drawOrderBlocks(bool internal = false) =>
    array<orderBlock> obs = internal ? internalOrderBlocks : swingOrderBlocks
    int n = obs.size()
    if n > 0
        int maxOBs = internal ? intOBCntInp : swOBCntInp
        array<orderBlock> shown = obs.slice(0, math.min(maxOBs, n))
        array<box> boxes        = internal ? internalOBBoxes : swingOBBoxes
        for [idx, ob] in shown
            color c = internal
                 ? (ob.bias == BEARISH ? intBearOB : intBullOB)
                 : (ob.bias == BEARISH ? swBearOB  : swBullOB)
            box b = boxes.get(idx)
            b.set_top_left_point(    chart.point.new(ob.barTime, na, ob.barHigh))
            b.set_bottom_right_point(chart.point.new(last_bar_time, na, ob.barLow))
            b.set_border_color(      internal ? na : c)
            b.set_bgcolor(           c)

// ════════════════════════════════════════════════════════════════════════════
// STRUCTURE DETECTION + DISPATCH
// ════════════════════════════════════════════════════════════════════════════
displayStructure(bool internal = false) =>
    bool bullishBar = true
    bool bearishBar = true
    if intConflInp
        bullishBar := high - math.max(close, open) > math.min(close, open - low)
        bearishBar := high - math.max(close, open) < math.min(close, open - low)

    pivot  p   = internal ? internalHigh : swingHigh
    trend  t   = internal ? internalTrend : swingTrend
    string ls  = internal ? line.style_dashed : line.style_solid
    string sz  = internal ? intLblSzInp : swLblSzInp
    string sztk= labelSize(sz)

    bool   extraBull = internal ? (internalHigh.currentLevel != swingHigh.currentLevel and bullishBar) : true
    color  bullBase  = internal ? intBullCol : swBullCol
    string bullFilt  = internal ? intBullFilterInp : swBullFilterInp
    bool   showCat   = internal ? showIntInp : showSwInp

    if ta.crossover(close, p.currentLevel) and not p.crossed and extraBull
        string tag = t.bias == BEARISH ? CHOCH_S : BOS_S
        if internal
            currentAlerts.intBullCHoCH := tag == CHOCH_S
            currentAlerts.intBullBOS   := tag == BOS_S
        else
            currentAlerts.swBullCHoCH  := tag == CHOCH_S
            currentAlerts.swBullBOS    := tag == BOS_S
        p.crossed := true
        t.bias    := BULLISH
        bool show = showCat and (bullFilt == ALL or (bullFilt == BOS_S and tag != CHOCH_S) or (bullFilt == CHOCH_S and tag == CHOCH_S))
        if show
            color c = strengthCol(bullBase, math.abs(close - p.currentLevel))
            drawStructure(p, tag, c, ls, label.style_label_down, sztk)
        if (internal and showIntOBInp) or (not internal and showSwOBInp)
            storeOrderBlock(p, internal, BULLISH)

    // bearish branch
    p             := internal ? internalLow : swingLow
    bool extraBear= internal ? (internalLow.currentLevel != swingLow.currentLevel and bearishBar) : true
    color bearBase= internal ? intBearCol : swBearCol
    string bearFilt = internal ? intBearFilterInp : swBearFilterInp

    if ta.crossunder(close, p.currentLevel) and not p.crossed and extraBear
        string tag = t.bias == BULLISH ? CHOCH_S : BOS_S
        if internal
            currentAlerts.intBearCHoCH := tag == CHOCH_S
            currentAlerts.intBearBOS   := tag == BOS_S
        else
            currentAlerts.swBearCHoCH  := tag == CHOCH_S
            currentAlerts.swBearBOS    := tag == BOS_S
        p.crossed := true
        t.bias    := BEARISH
        bool show = showCat and (bearFilt == ALL or (bearFilt == BOS_S and tag != CHOCH_S) or (bearFilt == CHOCH_S and tag == CHOCH_S))
        if show
            color c = strengthCol(bearBase, math.abs(close - p.currentLevel))
            drawStructure(p, tag, c, ls, label.style_label_up, sztk)
        if (internal and showIntOBInp) or (not internal and showSwOBInp)
            storeOrderBlock(p, internal, BEARISH)

// ════════════════════════════════════════════════════════════════════════════
// FAIR VALUE GAPS
// Note: when fvgTFInp > chart TF, request.security() with lookahead_on may
// repaint until the current HTF bar closes (TradingView limitation).
// ════════════════════════════════════════════════════════════════════════════
fvgBox(int leftTime, int rightTime, float topPrice, float bottomPrice, color c) =>
    box.new(
         chart.point.new(leftTime, na, topPrice),
         chart.point.new(rightTime + fvgExtInp * (time - time[1]), na, bottomPrice),
         xloc=xloc.bar_time, border_color=c, bgcolor=c)

deleteFairValueGaps() =>
    for [idx, fvg] in fairValueGaps
        if (low < fvg.bottom and fvg.bias == BULLISH) or (high > fvg.top and fvg.bias == BEARISH)
            fvg.topBox.delete()
            fvg.bottomBox.delete()
            fairValueGaps.remove(idx)

drawFairValueGaps() =>
    [lastClose, lastOpen, lastTime, curHigh, curLow, curTime, last2High, last2Low] =
         request.security(syminfo.tickerid, fvgTFInp,
              [close[1], open[1], time[1], high[0], low[0], time[0], high[2], low[2]],
              lookahead = barmerge.lookahead_on)

    float deltaPct = (lastClose - lastOpen) / (lastOpen * 100)
    bool  newTF    = timeframe.change(fvgTFInp)
    float thresh   = fvgAutoTInp ? ta.cum(math.abs(newTF ? deltaPct : 0.0)) / math.max(bar_index, 1) * 2 : 0.0

    bool bullFVG = curLow  > last2High and lastClose > last2High and  deltaPct > thresh and newTF
    bool bearFVG = curHigh < last2Low  and lastClose < last2Low  and -deltaPct > thresh and newTF

    if bullFVG
        currentAlerts.bullFVG := true
        float mid = math.avg(curLow, last2High)
        fairValueGaps.unshift(fairValueGap.new(curLow, last2High, BULLISH,
             fvgBox(lastTime, curTime, curLow,   mid,        fvgBullCol),
             fvgBox(lastTime, curTime, mid,      last2High,  fvgBullCol)))
    if bearFVG
        currentAlerts.bearFVG := true
        float mid = math.avg(curHigh, last2Low)
        fairValueGaps.unshift(fairValueGap.new(curHigh, last2Low, BEARISH,
             fvgBox(lastTime, curTime, curHigh,  mid,        fvgBearCol),
             fvgBox(lastTime, curTime, mid,      last2Low,   fvgBearCol)))

// ════════════════════════════════════════════════════════════════════════════
// MTF LEVELS
// ════════════════════════════════════════════════════════════════════════════
higherTimeframe(string tf) => timeframe.in_seconds() > timeframe.in_seconds(tf)

drawLevels(string tf, bool sameTF, string style, color col) =>
    [topLvl, botLvl, lTime, rTime] = request.security(syminfo.tickerid, tf,
         [high[1], low[1], time[1], time], lookahead=barmerge.lookahead_on)

    float parsedTop = sameTF ? high : topLvl
    float parsedBot = sameTF ? low  : botLvl
    int   pLeft     = sameTF ? time : lTime
    int   pRight    = sameTF ? time : rTime
    int   topTime   = time
    int   botTime   = time

    if not sameTF
        int lIdx = times.binary_search_rightmost(pLeft)
        int rIdx = times.binary_search_rightmost(pRight)
        array<int>   tArr  = times.slice(lIdx, rIdx)
        array<float> topA  = highs.slice(lIdx, rIdx)
        array<float> botA  = lows.slice(lIdx, rIdx)
        topTime := tArr.size() > 0 ? tArr.get(topA.indexof(topA.max())) : initialTime
        botTime := tArr.size() > 0 ? tArr.get(botA.indexof(botA.min())) : initialTime

    var line  topL = line.new(na,na,na,na, xloc=xloc.bar_time, color=col, style=lineStyle(style))
    var line  botL = line.new(na,na,na,na, xloc=xloc.bar_time, color=col, style=lineStyle(style))
    var label topB = label.new(na,na, xloc=xloc.bar_time,
         text=str.format('P{0}H', tf), color=color(na), textcolor=col,
         size=size.small, style=label.style_label_left)
    var label botB = label.new(na,na, xloc=xloc.bar_time,
         text=str.format('P{0}L', tf), color=color(na), textcolor=col,
         size=size.small, style=label.style_label_left)

    int rEnd = last_bar_time + 20 * (time - time[1])
    topL.set_first_point( chart.point.new(topTime, na, parsedTop))
    topL.set_second_point(chart.point.new(rEnd,    na, parsedTop))
    topB.set_point(       chart.point.new(rEnd,    na, parsedTop))
    botL.set_first_point( chart.point.new(botTime, na, parsedBot))
    botL.set_second_point(chart.point.new(rEnd,    na, parsedBot))
    botB.set_point(       chart.point.new(rEnd,    na, parsedBot))

// ════════════════════════════════════════════════════════════════════════════
// TRAILING SWINGS / PREMIUM-DISCOUNT
// ════════════════════════════════════════════════════════════════════════════
updateTrailingExtremes() =>
    trailing.top            := math.max(high, trailing.top)
    trailing.lastTopTime    := trailing.top    == high ? time : trailing.lastTopTime
    trailing.bottom         := math.min(low,  trailing.bottom)
    trailing.lastBottomTime := trailing.bottom == low  ? time : trailing.lastBottomTime

drawHighLowSwings() =>
    var line  topLn = line.new(na,na,na,na,  color=swBearCol, xloc=xloc.bar_time)
    var line  botLn = line.new(na,na,na,na,  color=swBullCol, xloc=xloc.bar_time)
    var label topLb = label.new(na,na, color=color(na), textcolor=swBearCol,
         xloc=xloc.bar_time, style=label.style_label_down, size=size.tiny)
    var label botLb = label.new(na,na, color=color(na), textcolor=swBullCol,
         xloc=xloc.bar_time, style=label.style_label_up,   size=size.tiny)

    int rEnd = last_bar_time + 20 * (time - time[1])
    topLn.set_first_point( chart.point.new(trailing.lastTopTime, na, trailing.top))
    topLn.set_second_point(chart.point.new(rEnd,                 na, trailing.top))
    topLb.set_point(       chart.point.new(rEnd,                 na, trailing.top))
    topLb.set_text(        swingTrend.bias == BEARISH ? 'Strong High' : 'Weak High')

    botLn.set_first_point( chart.point.new(trailing.lastBottomTime, na, trailing.bottom))
    botLn.set_second_point(chart.point.new(rEnd,                    na, trailing.bottom))
    botLb.set_point(       chart.point.new(rEnd,                    na, trailing.bottom))
    botLb.set_text(        swingTrend.bias == BULLISH ? 'Strong Low' : 'Weak Low')

drawZone(float lblLvl, int lblIdx, float top, float bot, string tag, color zCol, string style) =>
    var label l = label.new(na,na, text=tag, color=color(na), textcolor=zCol,
         style=style, size=size.small)
    var box   b = box.new(na,na,na,na, bgcolor=color.new(zCol, zoneT),
         border_color=color(na), xloc=xloc.bar_time)
    b.set_top_left_point(    chart.point.new(trailing.barTime,  na, top))
    b.set_bottom_right_point(chart.point.new(last_bar_time,     na, bot))
    l.set_point(             chart.point.new(na, lblIdx, lblLvl))

drawPremiumDiscountZones() =>
    drawZone(trailing.top, math.round(0.5 * (trailing.barIndex + last_bar_index)),
         trailing.top, 0.95 * trailing.top + 0.05 * trailing.bottom,
         'Premium', premCol, label.style_label_down)
    float eqLvl = math.avg(trailing.top, trailing.bottom)
    drawZone(eqLvl, last_bar_index,
         0.525 * trailing.top    + 0.475 * trailing.bottom,
         0.525 * trailing.bottom + 0.475 * trailing.top,
         'Equilibrium', eqCol, label.style_label_left)
    drawZone(trailing.bottom, math.round(0.5 * (trailing.barIndex + last_bar_index)),
         0.95 * trailing.bottom + 0.05 * trailing.top, trailing.bottom,
         'Discount', discCol, label.style_label_up)
