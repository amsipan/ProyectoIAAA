length      = input.int(50, 'Pivot Length', minval=2, group='Swing Points Extraction')
show_reg    = input.bool(true, 'Display Regular Pivots', inline='inline1', group='Swing Points Extraction')
reg_ph_css  = input.color(#ef5350, 'High', inline='inline1', group='Swing Points Extraction')
reg_pl_css  = input.color(#26a69a, 'Low', inline='inline1', group='Swing Points Extraction')

show_miss   = input.bool(true, 'Display Missed (Ghost) Pivots', inline='inline2', group='Swing Points Extraction')
miss_ph_css = input.color(#ef5350, 'High', inline='inline2', group='Swing Points Extraction')
miss_pl_css = input.color(#26a69a, 'Low', inline='inline2', group='Swing Points Extraction')
label_css   = input.color(color.white, 'Text Label Color', group='Swing Points Extraction')

// --- Rastro ---
rastro_css  = input.color(color.new(color.gray, 30), 'Color del Rastro (1)', group='Swing Points Extraction')

// Ordinary VWAP Price Inputs
priceSource = input.string("HLC3", "VWAP Price Source", options=["Close", "OC2", "HLC3", "OHLC4"], group='VWAP Calculation Settings')

// Standard Deviation Band Inputs
show_sd1    = input.bool(true, "Enable 1 Std Dev Bands", group="VWAP Standard Deviation Bands")
sd1_css     = input.color(color.new(color.green, 40), "1 SD Color", group="VWAP Standard Deviation Bands")
show_sd2    = input.bool(true, "Enable 2 Std Dev Bands (Default)", group="VWAP Standard Deviation Bands")
sd2_css     = input.color(color.new(color.orange, 30), "2 SD Color", group="VWAP Standard Deviation Bands")
show_sd3    = input.bool(false, "Enable 3 Std Dev Bands", group="VWAP Standard Deviation Bands")
sd3_css     = input.color(color.new(color.red, 40), "3 SD Color", group="VWAP Standard Deviation Bands")

highS       = input.color(color.lime, title="Swing Labels Up", group="Style", inline="Swing")
lowS        = input.color(color.red, title="Swing Labels Down", group="Style", inline="Swing")
S           = input.color(color.blue,  title="VWAP Line Up", group="Style", inline="VWAP")
R           = input.color(color.purple, title="VWAP Line Down", group="Style", inline="VWAP")
xx          = input.int(2, minval=1, title="VWAP Width", group="Style", inline="VWAP")

// ==========================================
// ~~ Structures & Native Global Variables

b = bar_index

type SwingPoint
    bool  found
    int   index
    float price
    int   direction // 1 for Low/Support, -1 for High/Resistance
    bool  is_ghost

type dataPoints
    array<chart.point> points
    polyline poly = na

type vwapBands
    array<chart.point> u1_pts
    array<chart.point> l1_pts
    array<chart.point> u2_pts
    array<chart.point> l2_pts
    array<chart.point> u3_pts
    array<chart.point> l3_pts
    polyline u1_poly = na
    polyline l1_poly = na
    polyline u2_poly = na
    polyline l2_poly = na
    polyline u3_poly = na
    polyline l3_poly = na

var vwapData = dataPoints.new(array.new<chart.point>())
var vwapBands bandsData = vwapBands.new(array.new<chart.point>(), array.new<chart.point>(), array.new<chart.point>(), array.new<chart.point>(), array.new<chart.point>(), array.new<chart.point>())

var dataPoints ghostVwapData = dataPoints.new(array.new<chart.point>())
var vwapBands ghostBandsData = vwapBands.new(array.new<chart.point>(), array.new<chart.point>(), array.new<chart.point>(), array.new<chart.point>(), array.new<chart.point>(), array.new<chart.point>())

// Persistent global reference for the live moving ghost label
var label live_ghost_label = na

// Original Global variables kept intact as standard serial states
var line zigzag = na
var line ghost_level = na
var float max = 0.
var float min = 0.
var int max_x1 = 0
var int min_x1 = 0
var float follow_max = 0.
var int follow_max_x1 = 0
var float follow_min = 0.
var int follow_min_x1 = 0
var int os = 0
var float py1 = 0.
var int px1 = 0

// ==========================================
// ~~ Native Pivot Extraction Engine (LuxAlgo)
// ==========================================
max := math.max(high[length], max)
min := math.min(low[length], min)
follow_max := math.max(high[length], follow_max)
follow_min := math.min(low[length], follow_min)

if max > max[1]
    max_x1 := b - length
    follow_min := low[length]
if min < min[1]
    min_x1 := b - length    
    follow_max := high[length]

if follow_min < follow_min[1]
    follow_min_x1 := b - length
if follow_max > follow_max[1]
    follow_max_x1 := b - length

line.set_x2(ghost_level[1], b)

// Container arrays to interact across scopes safely
var SwingPoint[] discovered_pivots = array.new<SwingPoint>()
var line[] line_refs = array.new<line>(2, na) // Index 0: zigzag, Index 1: ghost_level

get_swing_pivots(float ph, float pl, line[] lines, float _max, float _min, int _max_x1, int _min_x1, float _f_max, float _f_min, int _f_max_x1, int _f_min_x1, int _os, int _px1, float _py1) =>
    array.clear(discovered_pivots)
    
    l_zigzag      = array.get(lines, 0)
    l_ghost_level = array.get(lines, 1)

    int local_px1 = _px1
    local_py1     = _py1
    local_os      = _os
    local_max     = _max
    local_min     = _min

    if not na(ph)  
        if show_miss
            if _os == 1 
                label.new(_min_x1, _min, '👻', color=miss_pl_css, style=label.style_label_up, size=size.small, tooltip=str.tostring(_min, '#.####'))
                l_zigzag := line.new(local_px1, local_py1, _min_x1, _min, color=miss_ph_css, style=line.style_dashed)
                local_px1 := _min_x1, local_py1 := _min
                line.set_x2(l_ghost_level, local_px1)
                l_ghost_level := line.new(local_px1, local_py1, local_px1, local_py1, color=color.new(reg_pl_css, 50), width=2)
                
                array.push(discovered_pivots, SwingPoint.new(true, _min_x1, _min, 1, true))
                
            else if ph < _max
                label.new(_max_x1, _max, '👻', color=miss_ph_css, style=label.style_label_down, size=size.small, tooltip=str.tostring(_max, '#.####'))
                label.new(_f_min_x1, _f_min, '👻', color=miss_pl_css, style=label.style_label_up, size=size.small, tooltip=str.tostring(_min, '#.####'))
                
                l_zigzag := line.new(local_px1, local_py1, _max_x1, _max, color=miss_pl_css, style=line.style_dashed)
                local_px1 := _max_x1, local_py1 := _max
                line.set_x2(l_ghost_level, local_px1)
                l_ghost_level := line.new(local_px1, local_py1, local_px1, local_py1, color=color.new(reg_ph_css, 50), width=2)
                
                array.push(discovered_pivots, SwingPoint.new(true, _max_x1, _max, -1, true))
                
                l_zigzag := line.new(local_px1, local_py1, _f_min_x1, _f_min, color=miss_ph_css, style=line.style_dashed)
                local_px1 := _f_min_x1, local_py1 := _f_min
                line.set_x2(l_ghost_level, local_px1)
                l_ghost_level := line.new(local_px1, local_py1, local_px1, local_py1, color=color.new(reg_pl_css, 50), width=2)
                
                array.push(discovered_pivots, SwingPoint.new(true, _f_min_x1, _f_min, 1, true))

        if show_reg
            label.new(b - length, ph, '▼', textcolor=label_css, color=reg_ph_css, style=label.style_label_down, size=size.small, tooltip=str.tostring(ph, '#.####'))
            l_zigzag := line.new(local_px1, local_py1, b - length, ph, color=miss_pl_css, style=ph < _max or _os == 1 ? line.style_dashed : line.style_solid)
        
        array.push(discovered_pivots, SwingPoint.new(true, b - length, ph, -1, false))
        local_py1 := ph, local_px1 := b - length, local_os := 1, local_max := ph, local_min := ph

    if not na(pl)  
        if show_miss  
            if _os == 0 
                label.new(_max_x1, _max, '👻', color=miss_ph_css, style=label.style_label_down, size=size.small, tooltip=str.tostring(_max, '#.####'))
                l_zigzag := line.new(local_px1, local_py1, _max_x1, _max, color=miss_pl_css, style=line.style_dashed)
                local_px1 := _max_x1, local_py1 := _max
                line.set_x2(l_ghost_level, local_px1)
                l_ghost_level := line.new(local_px1, local_py1, local_px1, local_py1, color=color.new(reg_ph_css, 50), width=2)
                
                array.push(discovered_pivots, SwingPoint.new(true, _max_x1, _max, -1, true))
                
            else if pl > _min
                label.new(_f_max_x1, _f_max, '👻', color=miss_ph_css, style=label.style_label_down, size=size.small, tooltip=str.tostring(_max, '#.####'))
                label.new(_min_x1, _min, '👻', color=miss_pl_css, style=label.style_label_up, size=size.small, tooltip=str.tostring(_min, '#.####'))
                
                l_zigzag := line.new(local_px1, local_py1, _min_x1, _min, color=miss_ph_css, style=line.style_dashed)
                local_px1 := _min_x1, local_py1 := _min
                line.set_x2(l_ghost_level, local_px1)
                l_ghost_level := line.new(local_px1, local_py1, local_px1, local_py1, color=color.new(reg_pl_css, 50), width=2)
                
                array.push(discovered_pivots, SwingPoint.new(true, _min_x1, _min, 1, true))
                
                l_zigzag := line.new(local_px1, local_py1, _f_max_x1, _f_max, color=miss_pl_css, style=line.style_dashed)
                local_px1 := _f_max_x1, local_py1 := _f_max
                line.set_x2(l_ghost_level, local_px1)
                l_ghost_level := line.new(local_px1, local_py1, local_px1, local_py1, color=color.new(reg_ph_css, 50), width=2)
                
                array.push(discovered_pivots, SwingPoint.new(true, _f_max_x1, _f_max, -1, true))

        if show_reg
            label.new(b - length, pl, '▲', textcolor=label_css, color=reg_pl_css, style=label.style_label_up, size=size.small, tooltip=str.tostring(pl, '#.####'))
            l_zigzag := line.new(local_px1, local_py1, b - length, pl, color=miss_ph_css, style=pl > _min or _os == 0 ? line.style_dashed : line.style_solid)
        
        array.push(discovered_pivots, SwingPoint.new(true, b - length, pl, 1, false))
        local_py1 := pl, local_px1 := b - length, local_os := 0, local_max := pl, local_min := pl

    array.set(lines, 0, l_zigzag)
    array.set(lines, 1, l_ghost_level)
    
    [discovered_pivots, local_px1, local_py1, local_os, local_max, local_min]

// ==========================================
// ~~ Execution Frame & State Capturing
// ==========================================
_ph = ta.pivothigh(length, length)
_pl = ta.pivotlow(length, length)

array.set(line_refs, 0, zigzag)
array.set(line_refs, 1, ghost_level)

[pivots, new_px1, new_py1, new_os, new_max, new_min] = get_swing_pivots(_ph, _pl, line_refs, max[1], min[1], max_x1[1], min_x1[1], follow_max[1], follow_min[1], follow_max_x1[1], follow_min_x1[1], os[1], px1[1], py1[1])

zigzag      := array.get(line_refs, 0)
ghost_level := array.get(line_refs, 1)
if not na(_ph) or not na(_pl)
    px1 := new_px1, py1 := new_py1, os := new_os, max := new_max, min := new_min

// ==========================================
// ~~ Price Source Handler
// ==========================================
srcPrice = switch priceSource
    "Close" => close
    "OC2"   => (open + close) / 2
    "HLC3"  => (high + low + close) / 3
    "OHLC4" => (open + high + low + close) / 4
    => hlc3

// ==========================================
// ~~ Ordinary Dynamic VWAP Tracking Engine
// ==========================================
var int anchor_x = 0
var float anchor_y = 0.0
var int current_dir = 0
var float prev_pivot_price = float(na)

// Serial cumulative tracking states for the real-time continuous calculations
var float active_cumVol   = 0.0
var float active_cumPriceVol = 0.0
var float active_sumSqDiff   = 0.0

clean_band_polylines(vwapBands targetBands) =>
    targetBands.u1_poly.delete(), targetBands.l1_poly.delete()
    targetBands.u2_poly.delete(), targetBands.l2_poly.delete()
    targetBands.u3_poly.delete(), targetBands.l3_poly.delete()

clear_band_points(vwapBands targetBands) =>
    targetBands.u1_pts.clear(), targetBands.l1_pts.clear()
    targetBands.u2_pts.clear(), targetBands.l2_pts.clear()
    targetBands.u3_pts.clear(), targetBands.l3_pts.clear()

generate_band_polylines(vwapBands targetBands) =>
    if show_sd1
        targetBands.u1_poly := polyline.new(targetBands.u1_pts, false, false, line_color = sd1_css, line_width = 1)
        targetBands.l1_poly := polyline.new(targetBands.l1_pts, false, false, line_color = sd1_css, line_width = 1)
    if show_sd2
        targetBands.u2_poly := polyline.new(targetBands.u2_pts, false, false, line_color = sd2_css, line_width = 1)
        targetBands.l2_poly := polyline.new(targetBands.l2_pts, false, false, line_color = sd2_css, line_width = 1)
    if show_sd3
        targetBands.u3_poly := polyline.new(targetBands.u3_pts, false, false, line_color = sd3_css, line_width = 1)
        targetBands.l3_poly := polyline.new(targetBands.l3_pts, false, false, line_color = sd3_css, line_width = 1)

if array.size(pivots) > 0
    SwingPoint active_pivot = array.get(pivots, array.size(pivots) - 1)
    
    anchor_x    := active_pivot.index
    anchor_y    := active_pivot.price
    current_dir := active_pivot.direction
    
    //txt = current_dir > 0 and anchor_y < prev_pivot_price ? 'LL' : current_dir > 0 and anchor_y > prev_pivot_price ? 'HL' : current_dir < 0 and anchor_y < prev_pivot_price ? 'LH' : current_dir < 0 and anchor_y > prev_pivot_price ? 'HH' : ''
    //label.new(anchor_x, anchor_y, text=txt, style=current_dir > 0 ? label.style_label_up : label.style_label_down, color=color.new(current_dir > 0 ? highS : lowS, 20), textcolor=color.white)
    prev_pivot_price := anchor_y

    barsback = b - anchor_x

    // Delete past lines and arrays to rebuild cleanly across historical barsback period
    vwapData.poly.delete()
    clean_band_polylines(bandsData)
    vwapData.points.clear()
    clear_band_points(bandsData)

    // Temporary loop accumulation stores to execute strict Ordinary VWAP calculations
    float hist_cumVol = 0.0
    float hist_cumPriceVol = 0.0
    float hist_sumSqDiff = 0.0

    for i = barsback to 0 by 1
        float v_i = volume[i]
        float p_i = srcPrice[i]
        
        hist_cumVol      += v_i
        hist_cumPriceVol += p_i * v_i
        float curr_vwap   = hist_cumVol > 0 ? hist_cumPriceVol / hist_cumVol : na
        
        hist_sumSqDiff   += v_i * math.pow(p_i - curr_vwap, 2)
        float curr_stdDev = hist_cumVol > 0 ? math.sqrt(hist_sumSqDiff / hist_cumVol) : 0.0

        int current_bar_idx = b - i
        vwapData.points.push(chart.point.from_index(current_bar_idx, curr_vwap))
        
        // Push Standard Deviation Points
        bandsData.u1_pts.push(chart.point.from_index(current_bar_idx, curr_vwap + curr_stdDev))
        bandsData.l1_pts.push(chart.point.from_index(current_bar_idx, curr_vwap - curr_stdDev))
        bandsData.u2_pts.push(chart.point.from_index(current_bar_idx, curr_vwap + 2 * curr_stdDev))
        bandsData.l2_pts.push(chart.point.from_index(current_bar_idx, curr_vwap - 2 * curr_stdDev))
        bandsData.u3_pts.push(chart.point.from_index(current_bar_idx, curr_vwap + 3 * curr_stdDev))
        bandsData.l3_pts.push(chart.point.from_index(current_bar_idx, curr_vwap - 3 * curr_stdDev))

    // Lock in the historical state into active serial stores for continuous calculation moving forward
    active_cumVol      := hist_cumVol
    active_cumPriceVol := hist_cumPriceVol
    active_sumSqDiff   := hist_sumSqDiff

    vwapData.poly := polyline.new(vwapData.points, false, false, line_color = current_dir < 0 ? R : S, line_width = xx)
    generate_band_polylines(bandsData)

else if current_dir != 0 
    // Live update calculations on bars following the anchor initialization zone
    active_cumVol      += volume
    active_cumPriceVol += srcPrice * volume
    float live_vwap     = active_cumVol > 0 ? active_cumPriceVol / active_cumVol : na
    
    active_sumSqDiff   += volume * math.pow(srcPrice - live_vwap, 2)
    float live_stdDev   = active_cumVol > 0 ? math.sqrt(active_sumSqDiff / active_cumVol) : 0.0

    vwapData.poly.delete()
    clean_band_polylines(bandsData)
    
    vwapData.points.push(chart.point.from_index(b, live_vwap))
    bandsData.u1_pts.push(chart.point.from_index(b, live_vwap + live_stdDev))
    bandsData.l1_pts.push(chart.point.from_index(b, live_vwap - live_stdDev))
    bandsData.u2_pts.push(chart.point.from_index(b, live_vwap + 2 * live_stdDev))
    bandsData.l2_pts.push(chart.point.from_index(b, live_vwap - 2 * live_stdDev))
    bandsData.u3_pts.push(chart.point.from_index(b, live_vwap + 3 * live_stdDev))
    bandsData.l3_pts.push(chart.point.from_index(b, live_vwap - 3 * live_stdDev))

    vwapData.poly := polyline.new(vwapData.points, false, false, line_color = current_dir < 0 ? R : S, line_width = xx)
    generate_band_polylines(bandsData)

// ==========================================
// ~~ Live Floating Ghost Pivot & Dynamic VWAP Extension
// ==========================================
if barstate.islast
    label.delete(live_ghost_label)
    
    x_last = 0, y_last = 0.
    prices = array.new_float(0)
    prices_x = array.new_int(0)
    
    for i = 0 to b - px1 - 1
        array.push(prices, os == 1 ? low[i] : high[i])
        array.push(prices_x, b - i)
        
    if array.size(prices) > 0
        if os == 1
            y_last := array.min(prices)
            x_last := array.get(prices_x, array.indexof(prices, y_last))
        else
            y_last := array.max(prices)
            x_last := array.get(prices_x, array.indexof(prices, y_last))
            
        // if show_miss
        //     line.delete(line.new(px1, py1, x_last, y_last, color=os == 1 ? miss_ph_css : miss_pl_css, style=line.style_dashed)[1])
        //     live_ghost_label := label.new(x_last, y_last, '👻', color=os == 1 ? miss_pl_css : miss_ph_css, style=os == 1 ? label.style_label_up : label.style_label_down, size=size.small, tooltip=str.tostring(y_last, '#.####'))
        if show_miss // Rastro
            line.delete(line.new(px1, py1, x_last, y_last, color=os == 1 ? miss_ph_css : miss_pl_css, style=line.style_dashed)[1])
            // Dibujamos el rastro histórico usando un "1" en lugar del emoji fantasma persistente
            label.new(x_last, y_last, '1', textcolor=color.white, color=os == 1 ? miss_pl_css : miss_ph_css, style=os == 1 ? label.style_label_up : label.style_label_down, size=size.small)

        line.delete(line.new(x_last, y_last, b, y_last, color = color.new(os == 1 ? miss_ph_css : miss_pl_css, 50), width=2)[1])

        if show_miss
            ghost_barsback = b - x_last
            
            ghostVwapData.poly.delete()
            clean_band_polylines(ghostBandsData)
            ghostVwapData.points.clear()
            clear_band_points(ghostBandsData)
            
            float ghost_cumVol = 0.0
            float ghost_cumPriceVol = 0.0
            float ghost_sumSqDiff = 0.0
            
            for i = ghost_barsback to 0 by 1
                float g_vol = volume[i]
                float g_prc = srcPrice[i]
                
                ghost_cumVol      += g_vol
                ghost_cumPriceVol += g_prc * g_vol
                float g_vwap       = ghost_cumVol > 0 ? ghost_cumPriceVol / ghost_cumVol : na
                
                ghost_sumSqDiff   += g_vol * math.pow(g_prc - g_vwap, 2)
                float g_stdDev     = ghost_cumVol > 0 ? math.sqrt(ghost_sumSqDiff / ghost_cumVol) : 0.0
                
                int ghost_bar_idx = b - i
                ghostVwapData.points.push(chart.point.from_index(ghost_bar_idx, g_vwap))
                
                ghostBandsData.u1_pts.push(chart.point.from_index(ghost_bar_idx, g_vwap + g_stdDev))
                ghostBandsData.l1_pts.push(chart.point.from_index(ghost_bar_idx, g_vwap - g_stdDev))
                ghostBandsData.u2_pts.push(chart.point.from_index(ghost_bar_idx, g_vwap + 2 * g_stdDev))
                ghostBandsData.l2_pts.push(chart.point.from_index(ghost_bar_idx, g_vwap - 2 * g_stdDev))
                ghostBandsData.u3_pts.push(chart.point.from_index(ghost_bar_idx, g_vwap + 3 * g_stdDev))
                ghostBandsData.l3_pts.push(chart.point.from_index(ghost_bar_idx, g_vwap - 3 * g_stdDev))
        
            ghostVwapData.poly := polyline.new(ghostVwapData.points, false, false, line_color = os == 1 ? S : R, line_width = xx)
            generate_band_polylines(ghostBandsData)

            // Rastro del fantasma
            line.delete(line.new(px1, py1, x_last, y_last, color=os == 1 ? miss_ph_css : miss_pl_css, style=line.style_dashed)[1])
            
            // 1. RASTRO HISTÓRICO: Deja un "1" limpio (sin recuadro) solo cuando la vela cierra y el fantasma cambia de lugar
            if barstate.isconfirmed
                label.new(x_last, y_last, '1', textcolor=os == 1 ? miss_pl_css : miss_ph_css, style=label.style_none, size=size.small)

            // 2. FANTASMA EN VIVO: Se dibuja y se borra en cada tick para mantenerse en la punta actual
            var label live_ghost = na
            label.delete(live_ghost)
            live_ghost := label.new(x_last, y_last, '👻', color=os == 1 ? miss_pl_css : miss_ph_css, style=os == 1 ? label.style_label_up : label.style_label_down, size=size.small, tooltip=str.tostring(y_last, '#.####'))
